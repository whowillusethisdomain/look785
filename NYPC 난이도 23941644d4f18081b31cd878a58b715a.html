아래는 실제 nypc연습 문제다

물풍선이 폭발하면 십자가의 형태로 무한한 직선이 생성된다. 첫 줄에 점의 개수를 나타내는 정수 N이 주어진다. (1≤N≤100 0001≤N≤100000)이어지는 NN 개의 줄의 i번째 줄에는 i 번째 물풍선이 놓인 좌표를 나타내는 두 정수 xi와 yi가 공백으로 구분되어 주어진다. (1≤xi,yi≤1 000 000 0001≤xi,yi≤1000000000)

생성되는 직선의 개수를 구하라

#include <stdio.h>
#include <stdlib.h>

int cmp(const void *a, const void *b) {
    return (*(int *)a - *(int *)b);
}

int count_duplicates(int *arr, int n) {
    qsort(arr, n, sizeof(int), cmp);
    int count = 0;

    for (int i = 0; i < n; ) {
        int j = i + 1;
        while (j < n && arr[i] == arr[j]) j++;
        if (j - i >= 2) count++;
        i = j;
    }

    return count;
}

int main() {
    int T;
    scanf("%d", &T);

    int *X = malloc(sizeof(int) * T);
    int *Y = malloc(sizeof(int) * T);

    for (int i = 0; i < T; i++) {
        scanf("%d %d", &X[i], &Y[i]);
    }

    int x_dup = count_duplicates(X, T);
    int y_dup = count_duplicates(Y, T);

    printf("%d\n", x_dup + y_dup);

    free(X);
    free(Y);
    return 0;
}

위 문제의 해설은 x,y에 해당하는 배열을 만들고 x,y의 겹치는 값을 구하는 함수를 만들어 x, y에 한번씩 대입하여 더한다, 대부분의 코드와는 다르게 이문제는 저 겹치는 코드를 만드는게 핵심이다
